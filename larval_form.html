Submitting will write to the per-container Drive workbook and append a summary row to the centralized class sheet.

Connect to Google

One Apps Script handles both writes — to the per-container Drive folder and to the centralized class sheet. Deploy it once and paste the URL into both forms.

Step 1 — Create a Drive folder for container logs

1
Go to drive.google.com, create a folder called "Butterfly Container Logs Spring 2026". Copy the ID from the URL (the string after /folders/).

Step 2 — Get your centralized sheet ID

2
Open your centralized class Google Sheet. Copy the ID from the URL (the string between /d/ and /edit).

Step 3 — Create a standalone Apps Script

3
Go to script.google.com, click New project, delete any existing code, and paste the script below.

Step 4 — Paste this script

const FOLDER_ID = 'YOUR_FOLDER_ID_HERE'; const CENTRAL_SHEET_ID = 'YOUR_CENTRAL_SHEET_ID_HERE'; const LARVAL_TAB = 'Larval growth and development'; const ECLOSION_TAB = 'Eclosions'; function doPost(e) { try { const data = JSON.parse(e.postData.contents); return writeLarval(data); } catch(err) { return ContentService .createTextOutput(JSON.stringify({success: false, error: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } // ── Larval: write to container workbook + centralized sheet ─────────────── function writeLarval(data) { // 1. Per-container Drive workbook const folder = DriveApp.getFolderById(FOLDER_ID); const fileName = data.container; let ss; const files = folder.getFilesByName(fileName); if (files.hasNext()) { ss = SpreadsheetApp.open(files.next()); } else { ss = SpreadsheetApp.create(fileName); DriveApp.getFileById(ss.getId()).moveTo(folder); } // One tab per observation date let sheet = ss.getSheetByName(data.date); if (!sheet) { sheet = ss.insertSheet(data.date); const def = ss.getSheetByName('Sheet1'); if (def && def.getLastRow() === 0) ss.deleteSheet(def); } // Header block (matches template) sheet.getRange(1,1).setValue("Today's Date:"); sheet.getRange(1,2).setValue(data.date); sheet.getRange(1,3).setValue("Diet:"); sheet.getRange(1,4).setValue(data.diet); sheet.getRange(1,5).setValue("Temperature:"); sheet.getRange(1,6).setValue(data.temp + "°C"); sheet.getRange(2,1).setValue("Date hatched:"); sheet.getRange(2,2).setValue(data.hatched); sheet.getRange(2,3).setValue("No. caterpillars:");sheet.getRange(2,4).setValue(data.total); sheet.getRange(2,5).setValue("Observers:"); sheet.getRange(2,6).setValue(data.observers); sheet.getRange(3,1).setValue("Container:"); sheet.getRange(3,2).setValue(data.container); sheet.getRange(3,3).setValue("No. living:"); sheet.getRange(3,4).setValue(data.alive); sheet.getRange(3,5).setValue("Campus:"); sheet.getRange(3,6).setValue(data.campus); // Measurement headers row 4 sheet.getRange(4,1).setValue("Caterpillar length"); sheet.getRange(4,2).setValue("Caterpillar width"); sheet.getRange(4,1,1,2).setFontWeight('bold').setBackground('#d4e8d8'); // Individual rows if (data.caterpillars && data.caterpillars.length > 0) { const rows = data.caterpillars.map(c => [c.length || '', c.width || '']); sheet.getRange(5, 1, rows.length, 2).setValues(rows); } sheet.autoResizeColumns(1, 6); // 2. Centralized class sheet — append summary row const central = SpreadsheetApp.openById(CENTRAL_SHEET_ID); let larvalSheet = central.getSheetByName(LARVAL_TAB); if (!larvalSheet) larvalSheet = central.insertSheet(LARVAL_TAB); const totalStartedSheet = central.getSheetByName('Total_Started'); const totalStartedData = totalStartedSheet.getDataRange().getValues(); let totalStarted = ''; for (let i = 1; i < totalStartedData.length; i++) { if (totalStartedData[i][0] === data.container) { totalStarted = totalStartedData[i][1]; break; } } // Add header row if sheet is empty if (larvalSheet.getLastRow() === 0) { larvalSheet.appendRow([ data.date, data.campus, data.temp, data.diet, data.container, data.hatched, data.alive, data.deaths, data.pupations, data.avgLen, data.avgWid, data.pupaLen, data.observers, data.notes, totalStarted ]); larvalSheet.getRange(1,1,1,13).setFontWeight('bold').setBackground('#d4e8d8'); } larvalSheet.appendRow([ data.date, data.campus, data.temp, data.diet, data.container, data.hatched, data.alive, data.deaths, data.pupations, data.avgLen, data.avgWid, data.pupaLen, data.observers, data.notes ]); return ContentService .createTextOutput(JSON.stringify({success: true})) .setMimeType(ContentService.MimeType.JSON); } // ── Eclosion: write to centralized sheet Eclosions tab ────────────────── function writeEclosion(data) { const central = SpreadsheetApp.openById(CENTRAL_SHEET_ID); let sheet = central.getSheetByName(ECLOSION_TAB); if (!sheet) sheet = central.insertSheet(ECLOSION_TAB); if (sheet.getLastRow() === 0) { sheet.appendRow([ 'Date','Campus','Temperature','Diet', 'Eclosed_Today','Cumulative_Eclosed', 'Total_Pupated','Cumulative_Percent', 'Observer','Notes' ]); sheet.getRange(1,1,1,10).setFontWeight('bold').setBackground('#ede9fe'); } sheet.appendRow([ data.date, data.campus, data.temp, data.diet, data.eclosedToday, data.cumulativeEclosed, data.totalPupated, data.cumulativePercent, data.observer, data.notes ]); return ContentService .createTextOutput(JSON.stringify({success: true})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status: 'ok'})) .setMimeType(ContentService.MimeType.JSON); }
